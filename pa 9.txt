

#include <stdio.h>
#include <stdlib.h>
#include<pic18f4550.h>
/*
 * 
 */
void main(void){
    int temp,i,j;
    int num[]={10,2,5,1,6};
    for(i=0;i<5;i++)
    {
        for(j=i+1;j<5;j++)
        {
            if(num[i]>num[j])
             {
               temp=num[i];
               num[i]=num[j];
               num[j]=temp;
            }
        }
    }
}
